#len
#radius
#histo )( $1fff
#sig

:clearhisto
	'histo >r
	64 ( 1? )( 1- 0 r!+ ) drop rdrop ;

:+histo
	1 swap 'histo + c+! ;

:signal | n -- value
	sig + c@ ;

:in | i j value -- i j value
	over pick3 - +histo ;

:makehist
	0 ( len <? )(
		dup 1+ ( len <? )(
        	over signal over signal - abs radius <? ( in ) drop
			1+ ) drop
		1+ ) drop ;

:+2dhisto | clz frec --
	swap 8 << +
	1 swap 'histo + c+! ;

:make2dhist
	0 ( len <? )(
		dup 1+ ( len <? )(
        	over signal over signal - abs
        	clz over pick3 - +2dhisto
			1+ ) drop
		1+ ) drop ;

|------------------
#m1 #m1>
#m2 #m2>
#m3 #m3>

:filla
	m1 23 ( 1? )( 1- 65 rot c!+ swap ) drop 'm1> ! ;

:fillr
	m1 1000 ( 1? )( 1- rand 2 >> $f and rot c!+ swap ) drop 'm1> ! ;

:fillt
	m1
	5 ( 1? )( 1- rand 6 >> rot c!+ swap ) drop
	700 ( 1? )( 1- 65 rot c!+ 63 swap c!+ swap ) drop
	4 ( 1? )( 1- rand 6 >> rot c!+ swap ) drop
	0 over !
	'm1> ! ;

:dumpl |
	( over <? )( c@+ emit ) 2drop ;
:dumph
	( over <? )( c@+ $ff and "%h " print ) 2drop ;

:main
	mark
	here dup 'm1 !
	$ffff + dup dup 'm2> ! 'm2 !
	$ffff + dup dup 'm3> ! 'm3 !
	'here !
	fillt
	m1 'sig !
	m1> m1 - 'len !
	show clrscr
		m1> m1 - "%d bytes" print cr
		m1> m1 dumph cr
		m3> m3 - "%d bytes" print cr
        m3> m3 dumph cr
		m2> m2 - "%d bytes" print cr
		m2> m2 dumph cr

		'make2dhist <f1>
		'exit >esc<
		cminiflecha ;

: main ;
