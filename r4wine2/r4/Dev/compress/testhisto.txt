#inic
#endc

#hist1
#hist2
#hist3
#hist4

:ini
	mark
	here dup 'inic !
	"..\enwik8\enwik8" load
	dup 'endc !
	dup 'hist1 !
	$400 + dup 'hist2 !
	$400 + dup 'hist3 !
	$400 + dup 'hist4 !
	'here !
	;

:clearhist
	hist1 >r $400 2 << ( 1? )( 0 r!+ 1- ) drop rdrop ;

:makehisto
	inic ( endc <? )(
		@+
		1 over $ff and 2 << hist1 + +!
		6 >> 1 over $3fc and hist2 + +!
		8 >> 1 over $3fc and hist3 + +!
		8 >> 1 swap $3fc and hist4 + +!
		) drop ;

:hist
	clearhist
	makehisto
	;

:showline | pos dx hist -- pos
		pick2 2 << + @ 10 + >r | pos dx cnt
		over 2 << + 10 over r> op line ;


		azul 0 hist1 showline
		rojo 1 hist2 showline
		verde 2 hist3 showline
		violeta 3 hist4 showline

:showhist
	0? ( drop ; )
	>r
	azul
	0 ( 256 <? )(
		0 over 20 + op
		r@+ over 20 + line
		1+ ) drop
	rdrop
	;
#act
:main
	show clrscr
		endc inic - "%d bytes " print cr
		act showhist

		'hist <f1>
		[ hist1 'act ! ; ] <f2>
		[ hist2 'act ! ; ] <f3>
		[ hist3 'act ! ; ] <f4>
		[ hist4 'act ! ; ] <f5>
		'exit >esc<
		cminiflecha
		;

: ini main ;
