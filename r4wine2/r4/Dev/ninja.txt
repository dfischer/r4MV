:[mapa]	| x y -- dir
	swap -? ( 0 nip ) anchomapa >=? ( 0 nip )
	swap -? ( 0 nip ) altomapa >=? ( 0 nip )
	anchomapa * + 2 << 'mapa + ;

:dibujotile | n -- dib
|	2 =? ( drop 'acoin msec 5 >> $1c and + ; )
|	4 =? ( drop 'alava msec 6 >> $1c and + ; )
	$ff and 2 << 'bloques + @ ;

:drawtile | x y -- x y
	over xp escalap >> +
	over yp escalap >> +
	[mapa] @ 0? ( drop ; )
	pick2 escalap << sw 2/ + xp $1f and -
	pick2 escalap << sh 2/ + yp $1f and -
	rot dibujotile
	bmr.draw
	;

	anchop neg ( anchop <? )(
		altop neg ( altop <? )(
        	drawtile

		1+ ) drop ;
