:dot4 | col --
	dup px!+ dup px!+ dup px!+ dup px!+ sw 4 - px+!
	dup px!+ dup px!+ dup px!+ dup px!+ sw 4 - px+!
	dup px!+ dup px!+ dup px!+ dup px!+ sw 4 - px+!
	dup px!+ dup px!+ dup px!+ px!+
	sw 3* neg px+! ;

::zoomx4 | size x y x0 y0 --
	sw * + 2 << framev + >r
	setxy
	dup 1+ ( 1? )( 1-
		over 1+ ( 1? )( 1-
			r@+ dot4
			) drop
		sw pick2 - 2 << 4 -
		dup px+! r+
		) 2drop
	rdrop ;
